def cache
  return @cache if @cache
  @cache = begin
    Rails.cache
  rescue NameError
    ActiveSupport::Cache.lookup_store(*cache_store)
  end
end

def cache_store=(a)
  @cache_store = [a].flatten
  @cache = ActiveSupport::Cache.lookup_store(a) if @cache
end
